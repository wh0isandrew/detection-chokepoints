- **Liquid-generated JSON data block** in a `<script>` tag BEFORE the framework-graph.js load:

```liquid
<script>
window.__GRAPH_DATA = (function() {
  var nodes = [];
  var links = [];
  var tacticSet = {};
  var techniqueSet = {};
  {% for cp in site.data.chokepoints %}
  // Chokepoint node
  nodes.push({
    id: "cp-{{ cp._slug }}",
    type: "chokepoint",
    name: {{ cp.Name | jsonify }},
    group: {{ cp.Tactics[0] | slugify | jsonify }},
    vars: {{ cp.Variations | size }},
    maturity: {% if cp.Detections %}{{ cp.Detections | last | map: "Level" | first | downcase | jsonify }}{% else %}"research"{% endif %},
    url: "/chokepoints/{{ cp._slug }}/"
  });
  // Tactic nodes (deduplicated)
  {% for tactic in cp.Tactics %}
  if (!tacticSet[{{ tactic | slugify | jsonify }}]) {
    tacticSet[{{ tactic | slugify | jsonify }}] = true;
    nodes.push({ id: "tac-" + {{ tactic | slugify | jsonify }}, type: "tactic", name: {{ tactic | jsonify }}, group: {{ tactic | slugify | jsonify }} });
  }
  {% endfor %}
  // Technique nodes + links
  {% for mid in cp.MitreIds %}
  if (!techniqueSet[{{ mid | jsonify }}]) {
    techniqueSet[{{ mid | jsonify }}] = true;
    nodes.push({ id: {{ mid | jsonify }}, type: "technique", name: {{ cp.Techniques[forloop.index0] | jsonify }}, group: {{ cp.Tactics[0] | slugify | jsonify }} });
  }
  links.push({ source: "cp-{{ cp._slug }}", target: {{ mid | jsonify }} });
  {% endfor %}
  // Link techniques to tactics
  {% for tactic in cp.Tactics %}{% for mid in cp.MitreIds %}
  links.push({ source: {{ mid | jsonify }}, target: "tac-" + {{ tactic | slugify | jsonify }} });
  {% endfor %}{% endfor %}
  {% endfor %}
  return { nodes: nodes, links: links };
})();
</script>
```
